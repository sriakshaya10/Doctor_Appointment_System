patient table

CREATE TABLE patient (
    id INT AUTO_INCREMENT PRIMARY KEY,
    patient_name VARCHAR(50) NOT NULL,
    email VARCHAR(50) NOT NULL UNIQUE,
    phone VARCHAR(20),
    language VARCHAR(50),
    password VARCHAR(10) NOT NULL,
    gender ENUM('Male', 'Female', 'Other') NOT NULL,
    age INT NOT NULL,
    blood_group ENUM('A+', 'A-', 'B+', 'B-', 'O+', 'O-', 'AB+', 'AB-') NOT NULL
);


lab bookings (sumana)

CREATE TABLE hospital_labbookings (
    id INT AUTO_INCREMENT PRIMARY KEY,
    patient_name VARCHAR(50),
    test_type VARCHAR(50),
    hospital VARCHAR(100),
    date DATE
);

doctor details

CREATE TABLE doctordetails (
    id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    email VARCHAR(30) NOT NULL UNIQUE,
    password VARCHAR(10) NOT NULL,
    phone_number VARCHAR(15),
    language VARCHAR(50),
    profession VARCHAR(50),
    qualification varchar(100),
    experience VARCHAR(50),
    address varchar(255),
    mode ENUM('Offline', 'Online') NOT NULL
);

offline_appointments

CREATE TABLE offline_appointments (
    id INT AUTO_INCREMENT PRIMARY KEY,
    location VARCHAR(60) NOT NULL,
    hospital VARCHAR(50) NOT NULL,
    department VARCHAR(40) NOT NULL,
    doctor VARCHAR(30) NOT NULL,
    slot VARCHAR(15) NOT NULL,
    patient_name VARCHAR(50) NOT NULL,
    age INT NOT NULL,
    phone VARCHAR(15) NOT NULL,
    appointment_date DATE NOT NULL
);

booking slots(online appointment)

CREATE TABLE booking_slots (
    id INT AUTO_INCREMENT PRIMARY KEY,
    p_name VARCHAR(50) NOT NULL,
    p_age INT NOT NULL,
    session VARCHAR(50),
    slot_time TIME NOT NULL,
    doctor_name VARCHAR(50) NOT NULL,
    department VARCHAR(30),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

doctor ratings

CREATE TABLE doctor_ratings (
    id INT AUTO_INCREMENT PRIMARY KEY,
    doctor_name VARCHAR(50) NOT NULL,
    rating INT NOT NULL CHECK (rating BETWEEN 1 AND 5),
    feedback varchar(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

home lab bookings

CREATE TABLE home_lab_bookings (
    id INT AUTO_INCREMENT PRIMARY KEY,
    fullname VARCHAR(50) NOT NULL,
    contact VARCHAR(15) NOT NULL,
    testname VARCHAR(40) NOT NULL,
    testprice DECIMAL(10, 2) NOT NULL,
    date DATE NOT NULL,
    time TIME NOT NULL,
    address varchar(255) NOT NULL
);

prescription

create table prescriptions(
    id INT AUTO_INCREMENT PRIMARY KEY,
    appointmentID VARCHAR(50) NOT NULL,
    patient_name VARCHAR(100) NOT NULL,
    doctor_name VARCHAR(100) NOT NULL,
    appointment_date DATE NOT NULL,
    medicines TEXT NOT NULL,
    instructions  varchar(200) not null,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
